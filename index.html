// ファイル選択時のメイン処理 document.getElementById('fileIn').addEventListener('change', async (e) => { const files = Array.from(e.target.files); if (files.length === 0) return; for (const file of files) { const text = await file.text(); const fileName = file.name.toUpperCase(); // 1. ペア名の特定 (ファイル名からキーワードを探す) const pair = pairKeywords.find(k => fileName.includes(k)); if (!pair) { writeLog("不明なファイル: " + file.name); continue; } const isPivot = fileName.includes("PIVOT"); // 2. チャート枠がなければ新規作成 let chart = chartMap.get(pair); if (!chart) { chart = new FXChart(pair); chartMap.set(pair, chart); } // 3. データの解析と格納 if (isPivot) { chart.dataPivot = parsePivotCSV(text); // [cite: 156, 160] writeLog(`${pair} Pivot 読み込み: ${chart.dataPivot.length}件`); } else { chart.dataH1 = parseCSV(text); // [cite: 151, 155] // SMAの計算などを実行 chart.dataSMA = getSMA(chart.dataH1, 10, 'v'); // [cite: 161] chart.dataPriceSMA = getSMA(chart.dataH1, 10, 'p'); writeLog(`${pair} H1 読み込み: ${chart.dataH1.length}件`); } // 4. 強制描画 chart.draw(); } });
